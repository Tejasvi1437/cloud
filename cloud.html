from flask import Flask, render_template_string, request, jsonify app = Flask(__name__) # A list to hold tasks (you could also store it in a database) tasks = [] # HTML template HTML_TEMPLATE = """ Beautiful To-Do List

To-Do List

    {% for task in tasks %}
  • {{ task['text'] }}
    • {% endfor %}
""" @app.route('/') def index(): return render_template_string(HTML_TEMPLATE, tasks=tasks) @app.route('/add', methods=['POST']) def add_task(): task_text = request.form.get('task') if task_text: tasks.append({'text': task_text, 'completed': False}) return jsonify(tasks=tasks) @app.route('/remove/', methods=['DELETE']) def remove_task(task_id): if 0 <= task_id < len(tasks): tasks.pop(task_id) return jsonify(tasks=tasks) @app.route('/toggle/', methods=['POST']) def toggle_task(task_id): if 0 <= task_id < len(tasks): tasks[task_id]['completed'] = not tasks[task_id]['completed'] return jsonify(tasks=tasks) if __name__ == '__main__': app.run(debug=True)
